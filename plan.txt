<script>
var currentCity = 'Mumbai';
var currentArea = 'Kurla';

function switchTab(tab, el) {
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
  if (tab === 'analytics') sendPrompt('Show me the analytics dashboard with person count trends, vehicle count trends, and watchlist match statistics');
  if (tab === 'investigate') sendPrompt('Open the investigation panel for cross-camera tracking and ANPR search');
}

function goToScreen(id) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('visible'));
  var el = document.getElementById(id);
  if (el) el.classList.add('visible');
  document.querySelectorAll('.nav-icon').forEach(n => n.classList.remove('active'));
  if (id === 'map-screen') document.querySelectorAll('.nav-icon')[0].classList.add('active');
  if (id === 'cam-screen') document.querySelectorAll('.nav-icon')[1].classList.add('active');
}

function openCity(city) {
  currentCity = city;
  document.getElementById('city-name-label').textContent = city;
  document.getElementById('city-back-label').textContent = city;
  goToScreen('city-screen');
}

function openArea(area) {
  currentArea = area;
  document.getElementById('area-name-label').textContent = area;
  goToScreen('cam-screen');
}

function askGenih() {
  var q = document.getElementById('genih-input').value.trim();
  if (!q) return;
  sendPrompt('GENIH query for ' + currentCity + ' / ' + currentArea + ': ' + q);
}
</script>
